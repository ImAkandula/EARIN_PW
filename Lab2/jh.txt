        process_t *child;

        if (depth > maxHeight) {
                maxHeight = depth;
                countAtMaxHeight = 1;
        }
        else if (depth == maxHeight) {
                countAtMaxHeight++;
        }

        // Iterate through children
        child = proc->children;
        while (child != NULL) {
                calculateHeightAndCount(child, depth + 1);
                child = child->next;
        }









        process_t *initProc;

        maxHeight = 0;
        countAtMaxHeight = 0;

        initProc = proc_find(1);  // PID 1 is usually init
        if (initProc == NULL)
                return -EINVAL;

        syscalls_calculateHeightAndCount(initProc, 0);

        if (proc_copytoUser(pHeight, &maxHeight, sizeof(int)) != 0 ||
                proc_copytoUser(pCount, &countAtMaxHeight, sizeof(int)) != 0)
                return -EFAULT;

        return 0;
